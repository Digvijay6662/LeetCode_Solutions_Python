Given an integer array nums, return the maximum possible sum of elements of the array such that it is divisible by three.

 

Example 1:

Input: nums = [3,6,5,1,8]
Output: 18
Explanation: Pick numbers 3, 6, 1 and 8 their sum is 18 (maximum sum divisible by 3).
Example 2:

Input: nums = [4]
Output: 0
Explanation: Since 4 is not divisible by 3, do not pick any number.
Example 3:

Input: nums = [1,2,3,4,4]
Output: 12
Explanation: Pick numbers 1, 3, 4 and 4 their sum is 12 (maximum sum divisible by 3).
 

Constraints:

1 <= nums.length <= 4 * 104
1 <= nums[i] <= 104


Solution 1:

class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        n = len(nums)

        # Memoization dictionary: key = (index, remainder), value = max sum achievable
        dp = {}

        def dfs(index, remainder):
            """
            Recursive function to compute the maximum sum divisible by 3
            starting from position 'index' with current remainder 'remainder'.
            """
            # Base case: reached end of array
            if index == n:
                # Valid sum only if remainder == 0
                return 0 if remainder == 0 else float('-inf')

            # If already computed, return stored result
            if (index, remainder) in dp:
                return dp[(index, remainder)]

            # Option 1: skip current number
            skip = dfs(index + 1, remainder)

            # Option 2: pick current number
            new_remainder = (remainder + nums[index]) % 3
            take = nums[index] + dfs(index + 1, new_remainder)

            # Store best result in dp
            dp[(index, remainder)] = max(skip, take)
            return dp[(index, remainder)]

        # Start recursion from index 0 with remainder 0
        result = dfs(0, 0)

        return result


Solution 2:

class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        n = len(nums)

        # dp[i][r] = maximum sum achievable using elements from index i...n-1
        # with remainder r (mod 3). We'll build this table iteratively.
        dp = [[float('-inf')] * 3 for _ in range(n + 1)]

        # Base case: at position n (past the last element), only remainder 0 is valid with sum 0
        dp[n][0] = 0

        # Fill table backwards from index n-1 down to 0
        for i in range(n - 1, -1, -1):
            for r in range(3):
                # Option 1: skip nums[i], remainder stays the same
                skip = dp[i + 1][r]

                # Option 2: take nums[i], remainder changes
                new_r = (r + nums[i]) % 3
                take = nums[i] + dp[i + 1][new_r]

                # Best of both choices
                dp[i][r] = max(skip, take)

        # Answer is the best sum starting at index 0 with remainder 0
        return dp[0][0]

class Solution:
    def maxSumDivThree(self, nums: List[int]) -> int:
        # dp[r] = maximum sum achievable with remainder r (mod 3)
        # Initialize: only sum 0 with remainder 0 is valid at start
        dp = [0, float('-inf'), float('-inf')]

        # Process each number in the array
        for num in nums:
            # Take a snapshot of current dp state before updating
            prev = dp[:]

            # Try adding 'num' to each possible remainder state
            for r in range(3):
                # New remainder if we add 'num' to a sum with remainder r
                new_r = (r + num) % 3

                # Update dp[r]:
                # - Option 1: keep current dp[r] (skip num)
                # - Option 2: add num to a sum with remainder new_r (take num)
                dp[r] = max(dp[r], prev[new_r] + num)

        # Final answer: maximum sum divisible by 3 (remainder 0)
        return dp[0]
