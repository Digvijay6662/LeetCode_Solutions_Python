You are given a string s consisting only of the characters 'a', 'b', and 'c'.

A substring of s is called balanced if all distinct characters in the substring appear the same number of times.

Return the length of the longest balanced substring of s.

 

Example 1:

Input: s = "abbac"

Output: 4

Explanation:

The longest balanced substring is "abba" because both distinct characters 'a' and 'b' each appear exactly 2 times.

Example 2:

Input: s = "aabcc"

Output: 3

Explanation:

The longest balanced substring is "abc" because all distinct characters 'a', 'b' and 'c' each appear exactly 1 time.

Example 3:

Input: s = "aba"

Output: 2

Explanation:

One of the longest balanced substrings is "ab" because both distinct characters 'a' and 'b' each appear exactly 1 time. Another longest balanced substring is "ba".

 

Constraints:

1 <= s.length <= 105
s contains only the characters 'a', 'b', and 'c'.


Solution 1:

class Solution:
    def longestBalanced(self, s: str) -> int:
        n = len(s)
        # Base case: Any single character is technically 'balanced' with itself.
        maxLength = 1   
        count = 1       

        # --- STEP 1: Single Character Substrings ---
        # A substring like "aaaa" is balanced because its only distinct character 'a' 
        # appears the same number of times (4). This finds the longest streak.
        for i in range(1, n):
            if s[i - 1] == s[i]:
                count += 1
            else:
                count = 1
            maxLength = max(maxLength, count)

        # --- STEP 2: Two Character Substrings ---
        # Logic: If (count1_at_j - count2_at_j) == (count1_at_i - count2_at_i),
        # then in the range (i, j], count1 and count2 increased by the same amount.
        def helper(ch1, ch2):
            count1 = count2 = 0
            # prefix maps the difference (count1 - count2) to the first index it was seen.
            # We initialize with {0: -1} to handle balanced substrings starting from index 0.
            prefix = {0: -1}   
            ans = 0

            for i in range(n):
                if s[i] == ch1:
                    count1 += 1
                elif s[i] == ch2:
                    count2 += 1
                else:
                    # If we hit the 3rd character (e.g., 'c' while checking 'a' & 'b'),
                    # the substring is no longer "balanced" by two characters.
                    # We reset the tracking to restart from the next index.
                    prefix = {0: i} 
                    count1 = count2 = 0
                    continue

                diff = count1 - count2
                if diff in prefix:
                    # Current index i minus the first index where this diff was seen.
                    ans = max(ans, i - prefix[diff])
                else:
                    prefix[diff] = i
            return ans

        # Check all possible pairs: {a,b}, {a,c}, {b,c}
        maxLength = max(maxLength, helper('a', 'b'))
        maxLength = max(maxLength, helper('a', 'c'))
        maxLength = max(maxLength, helper('b', 'c'))

        # --- STEP 3: Three Character Substrings ---
        # Logic: We need count_a == count_b == count_c.
        # This is true if (count_a - count_b) == 0 AND (count_a - count_c) == 0.
        # We store the tuple of differences as a unique key in our map.
        count_a = count_b = count_c = 0
        # Initialize with (0,0) at index -1 to catch balance from the very start.
        prefix = {(0, 0): -1}   

        for i in range(n):
            if s[i] == 'a':
                count_a += 1
            elif s[i] == 'b':
                count_b += 1
            else:
                count_c += 1

            # The state is defined by the relative differences between characters.
            key = (count_a - count_b, count_a - count_c)

            if key in prefix:
                # If we've seen this exact "difference state" before, 
                # the characters added since then are perfectly balanced.
                maxLength = max(maxLength, i - prefix[key])
            else:
                prefix[key] = i

        return maxLength
