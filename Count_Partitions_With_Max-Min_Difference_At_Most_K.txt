You are given an integer array nums and an integer k. Your task is to partition nums into one or more non-empty contiguous segments such that in each segment, the difference between its maximum and minimum elements is at most k.

Return the total number of ways to partition nums under this condition.

Since the answer may be too large, return it modulo 109 + 7.

 

Example 1:

Input: nums = [9,4,1,3,7], k = 4

Output: 6

Explanation:

There are 6 valid partitions where the difference between the maximum and minimum elements in each segment is at most k = 4:

[[9], [4], [1], [3], [7]]
[[9], [4], [1], [3, 7]]
[[9], [4], [1, 3], [7]]
[[9], [4, 1], [3], [7]]
[[9], [4, 1], [3, 7]]
[[9], [4, 1, 3], [7]]
Example 2:

Input: nums = [3,3,4], k = 0

Output: 2

Explanation:

There are 2 valid partitions that satisfy the given conditions:

[[3], [3], [4]]
[[3, 3], [4]]
 

Constraints:

2 <= nums.length <= 5 * 104
1 <= nums[i] <= 109
0 <= k <= 109


Solution 1:

class Solution:
    def countPartitions(self, nums: List[int], k: int) -> int:
        n = len(nums)
        MOD = 10**9 + 7
        # ways[i] = number of valid partitions for the first i elements
        ways = [0] * (n + 1)
        # prefix_sum[i] = cumulative sum of ways up to i (for fast range queries)
        prefix_sum = [0] * (n + 1)
        # Base case: empty array has exactly 1 valid partition (do nothing)
        ways[0] = prefix_sum[0] = 1
        # Left pointer for sliding window
        left = 0
        # Deques to maintain min and max in the current window
        min_deque, max_deque = deque(), deque()

        for right in range(n):
            # Maintain decreasing deque for maximum
            while max_deque and nums[max_deque[-1]] < nums[right]:
                max_deque.pop()
            max_deque.append(right)
            # Maintain increasing deque for minimum
            while min_deque and nums[min_deque[-1]] > nums[right]:
                min_deque.pop()
            min_deque.append(right)
            # Shrink window until condition (max - min <= k) is satisfied
            while left <= right and nums[max_deque[0]] - nums[min_deque[0]] > k:
                left += 1
                # Remove outdated indices from deques
                if min_deque and min_deque[0] < left:
                    min_deque.popleft()
                if max_deque and max_deque[0] < left:
                    max_deque.popleft()
            # Number of partitions ending at position `right`
            # = sum of ways for all valid split points between [left, right]
            ways[right + 1] = (prefix_sum[right] - (prefix_sum[left - 1] if left > 0 else 0)) % MOD
            # Update prefix sum for fast range queries
            prefix_sum[right + 1] = (prefix_sum[right] + ways[right + 1]) % MOD
        # Final answer: number of ways to partition the entire array
        return ways[n]
