You are given a string s of even length consisting of digits from 0 to 9, and two integers a and b.

You can apply either of the following two operations any number of times and in any order on s:

Add a to all odd indices of s (0-indexed). Digits post 9 are cycled back to 0. For example, if s = "3456" and a = 5, s becomes "3951".
Rotate s to the right by b positions. For example, if s = "3456" and b = 1, s becomes "6345".
Return the lexicographically smallest string you can obtain by applying the above operations any number of times on s.

A string a is lexicographically smaller than a string b (of the same length) if in the first position where a and b differ, string a has a letter that appears earlier in the alphabet than the corresponding letter in b. For example, "0158" is lexicographically smaller than "0190" because the first position they differ is at the third letter, and '5' comes before '9'.

Solution 1:

class Solution:
    def findLexSmallestString(self, s: str, a: int, b: int) -> str:
        visited = set()              # To keep track of already seen strings (avoid infinite loops)
        queue = deque([s])           # BFS queue initialized with the starting string
        smallest = s                 # Keep track of the lexicographically smallest string found so far

        while queue:
            current = queue.popleft()   # Get the next string to process

            if current in visited:      # Skip if we've already processed this string
                continue
            visited.add(current)        # Mark this string as visited

            # Update the smallest string if the current one is lexicographically smaller
            smallest = min(smallest, current)

            # --- Operation 1: Add 'a' to all digits at odd indices (0-indexed) ---
            chars = list(current)       # Convert string to list for mutability
            for i in range(1, len(chars), 2):   # Only odd indices
                chars[i] = str((int(chars[i]) + a) % 10)  # Add 'a' and wrap around using modulo 10
            added_version = "".join(chars)      # New string after addition

            # --- Operation 2: Rotate the string to the right by 'b' positions ---
            rotated_version = current[-b:] + current[:-b]

            # Push the new states into the queue for further exploration
            queue.append(added_version)
            queue.append(rotated_version)

        return smallest
