You are given an integer array nums of length n.

A trionic subarray is a contiguous subarray nums[l...r] (with 0 <= l < r < n) for which there exist indices l < p < q < r such that:

nums[l...p] is strictly increasing,
nums[p...q] is strictly decreasing,
nums[q...r] is strictly increasing.
Return the maximum sum of any trionic subarray in nums.

 

Example 1:

Input: nums = [0,-2,-1,-3,0,2,-1]

Output: -4

Explanation:

Pick l = 1, p = 2, q = 3, r = 5:

nums[l...p] = nums[1...2] = [-2, -1] is strictly increasing (-2 < -1).
nums[p...q] = nums[2...3] = [-1, -3] is strictly decreasing (-1 > -3)
nums[q...r] = nums[3...5] = [-3, 0, 2] is strictly increasing (-3 < 0 < 2).
Sum = (-2) + (-1) + (-3) + 0 + 2 = -4.
Example 2:

Input: nums = [1,4,2,7]

Output: 14

Explanation:

Pick l = 0, p = 1, q = 2, r = 3:

nums[l...p] = nums[0...1] = [1, 4] is strictly increasing (1 < 4).
nums[p...q] = nums[1...2] = [4, 2] is strictly decreasing (4 > 2).
nums[q...r] = nums[2...3] = [2, 7] is strictly increasing (2 < 7).
Sum = 1 + 4 + 2 + 7 = 14.
 

Constraints:

4 <= n = nums.length <= 105
-109 <= nums[i] <= 109
It is guaranteed that at least one trionic subarray exists.


Solution 1:

class Solution:
    def maxSumTrionic(self, nums: List[int]) -> int:
        n = len(nums)

        # Manual memoization dictionary: key = (index, phase), value = best sum
        dp = {}

        def dfs(index, phase):
            """
            Recursive function to compute the maximum sum of a trionic subarray
            starting at position `index` with the current phase `phase`.

            Phases:
              0 = before starting (haven't picked anything yet)
              1 = strictly increasing (first segment)
              2 = strictly decreasing (middle segment)
              3 = strictly increasing again (final segment)
            """

            # Base case: reached end of array
            if index == n:
                # Only valid if we have completed all 3 phases
                return 0 if phase == 3 else float('-inf')

            # Check memoized result
            if (index, phase) in dp:
                return dp[(index, phase)]

            current_val = nums[index]
            best_sum = float('-inf')

            # Option 1: skip current element
            # Skipping is only allowed before starting (phase 0)
            if phase == 0:
                best_sum = max(best_sum, dfs(index + 1, 0))

            # Option 2: take current element depending on phase
            if phase == 3:
                # In final increasing phase, we can always take current element
                # (this allows ending the sequence here)
                best_sum = max(best_sum, current_val)

            # If there is a next element, check transitions
            if index + 1 < n:
                next_val = nums[index + 1]

                if phase == 0 and current_val < next_val:
                    # Start first increasing phase
                    best_sum = max(best_sum, current_val + dfs(index + 1, 1))

                elif phase == 1:
                    if current_val < next_val:
                        # Continue increasing
                        best_sum = max(best_sum, current_val + dfs(index + 1, 1))
                    elif current_val > next_val:
                        # Switch to decreasing phase
                        best_sum = max(best_sum, current_val + dfs(index + 1, 2))

                elif phase == 2:
                    if current_val > next_val:
                        # Continue decreasing
                        best_sum = max(best_sum, current_val + dfs(index + 1, 2))
                    elif current_val < next_val:
                        # Switch to final increasing phase
                        best_sum = max(best_sum, current_val + dfs(index + 1, 3))

                elif phase == 3 and current_val < next_val:
                    # Continue final increasing phase
                    best_sum = max(best_sum, current_val + dfs(index + 1, 3))

            # Memoize result
            dp[(index, phase)] = best_sum
            return best_sum

        # Start recursion from index 0, phase 0 (before starting)
        return dfs(0, 0)

class Solution:
    def maxSumTrionic(self, nums: List[int]) -> int:
        n = len(nums)

        # Initialize DP table with -inf
        dp = [[float('-inf')] * 4 for _ in range(n + 1)]

        # Base case: at the end of the array, only phase 3 is valid
        dp[n][3] = 0

        # Fill DP table backwards
        for i in range(n - 1, -1, -1):
            curr = nums[i]
            for phase in range(4):
                best = float('-inf')

                # Skip option (only allowed before starting)
                if phase == 0:
                    best = max(best, dp[i + 1][0])

                # If in final increasing phase, can always take current
                if phase == 3:
                    best = max(best, curr)  # end here

                if i + 1 < n:
                    nxt = nums[i + 1]

                    if phase == 0 and curr < nxt:
                        # Start first increasing
                        best = max(best, curr + dp[i + 1][1])

                    elif phase == 1:
                        if curr < nxt:  # continue increasing
                            best = max(best, curr + dp[i + 1][1])
                        elif curr > nxt:  # switch to decreasing
                            best = max(best, curr + dp[i + 1][2])

                    elif phase == 2:
                        if curr > nxt:  # continue decreasing
                            best = max(best, curr + dp[i + 1][2])
                        elif curr < nxt:  # switch to final increasing
                            best = max(best, curr + dp[i + 1][3])

                    elif phase == 3 and curr < nxt:
                        # continue final increasing
                        best = max(best, curr + dp[i + 1][3])

                dp[i][phase] = best

        # Answer is starting at index 0, phase 0
        return dp[0][0]

class Solution:
    def maxSumTrionic(self, nums: List[int]) -> int:
        n = len(nums)

        # dp_next[phase] = best sum starting at i+1 with given phase
        dp_next = [float('-inf')] * 4
        dp_next[3] = 0  # base case: at end, only phase 3 is valid

        # Iterate backwards
        for i in range(n - 1, -1, -1):
            curr_val = nums[i]
            dp_curr = [float('-inf')] * 4

            for phase in range(4):
                best = float('-inf')

                # Skip option (only allowed before starting)
                if phase == 0:
                    best = max(best, dp_next[0])

                # If in final increasing phase, can always take current
                if phase == 3:
                    best = max(best, curr_val)  # end here

                if i + 1 < n:
                    next_val = nums[i + 1]

                    if phase == 0 and curr_val < next_val:
                        # Start first increasing
                        best = max(best, curr_val + dp_next[1])

                    elif phase == 1:
                        if curr_val < next_val:  # continue increasing
                            best = max(best, curr_val + dp_next[1])
                        elif curr_val > next_val:  # switch to decreasing
                            best = max(best, curr_val + dp_next[2])

                    elif phase == 2:
                        if curr_val > next_val:  # continue decreasing
                            best = max(best, curr_val + dp_next[2])
                        elif curr_val < next_val:  # switch to final increasing
                            best = max(best, curr_val + dp_next[3])

                    elif phase == 3 and curr_val < next_val:
                        # continue final increasing
                        best = max(best, curr_val + dp_next[3])

                dp_curr[phase] = best

            # Move current row to next row for next iteration
            dp_next = dp_curr

        # Answer is starting at index 0, phase 0
        return dp_next[0]

class Solution:
    def maxSumTrionic(self, nums: List[int]) -> int:
        n = len(nums)
        max_total_sum = float("-inf")   # global maximum sum across all trionic subarrays
        start_index = 0                 # pointer to scan the array

        while start_index < n:
            # Step 1: Find the first strictly increasing segment
            end_first = start_index + 1
            while end_first < n and nums[end_first - 1] < nums[end_first]:
                end_first += 1
            peak_index = end_first - 1  # last index of the first increasing segment

            # If no valid increasing segment (length < 2), move forward
            if peak_index == start_index:
                start_index += 1
                continue

            # Step 2: Find the strictly decreasing segment
            current_sum = nums[peak_index] + nums[peak_index - 1]  # include last two of first segment
            end_second = end_first
            while end_second < n and nums[end_second - 1] > nums[end_second]:
                current_sum += nums[end_second]
                end_second += 1
            valley_index = end_second - 1  # last index of decreasing segment

            # Validate decreasing segment
            if valley_index == peak_index or valley_index == n - 1 or (end_second < n and nums[end_second] <= nums[valley_index]):
                # If decreasing segment is invalid, skip ahead
                start_index = valley_index
                continue

            # Step 3: Start the third strictly increasing segment
            current_sum += nums[valley_index + 1]  # include first element of third segment

            # Extend the third segment to maximize sum
            max_right_extension = 0
            right_running_sum = 0
            k = valley_index + 2
            while k < n and nums[k] > nums[k - 1]:
                right_running_sum += nums[k]
                max_right_extension = max(max_right_extension, right_running_sum)
                k += 1
            current_sum += max_right_extension

            # Step 4: Extend the first segment backwards to maximize sum
            max_left_extension = 0
            left_running_sum = 0
            for k in range(peak_index - 2, start_index - 1, -1):
                left_running_sum += nums[k]
                max_left_extension = max(max_left_extension, left_running_sum)
            current_sum += max_left_extension

            # Step 5: Update global maximum
            max_total_sum = max(max_total_sum, current_sum)

            # Move start_index forward to continue scanning
            start_index = valley_index

        return max_total_sum
