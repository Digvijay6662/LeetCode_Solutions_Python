class Solution:
    def minimumTeachings(self, n: int, languages: List[List[int]], friendships: List[List[int]]) -> int:
        # Set of users who cannot currently communicate with at least one of their friends
        users_needing_help = set()

        # Step 1: Identify friendships where users cannot communicate
        for user_a, user_b in friendships:
            # Convert to 0-based indexing
            user_a -= 1
            user_b -= 1

            # Languages known by user_a
            langs_a = set(languages[user_a])

            # Check if they share at least one language
            can_communicate = False
            for lang in languages[user_b]:
                if lang in langs_a:
                    can_communicate = True
                    break

            # If they cannot communicate, mark both users as needing help
            if not can_communicate:
                users_needing_help.add(user_a)
                users_needing_help.add(user_b)

        # Step 2: Find the language that is already known by the maximum number of these users
        language_count = defaultdict(int)
        max_known_language_count = 0

        for user in users_needing_help:
            for lang in languages[user]:
                language_count[lang] += 1
                max_known_language_count = max(max_known_language_count, language_count[lang])

        # Step 3: Minimum users to teach = total affected users - those who already know the best language
        return len(users_needing_help) - max_known_language_count
