You are given an integer side, representing the edge length of a square with corners at (0, 0), (0, side), (side, 0), and (side, side) on a Cartesian plane.

You are also given a positive integer k and a 2D integer array points, where points[i] = [xi, yi] represents the coordinate of a point lying on the boundary of the square.

You need to select k elements among points such that the minimum Manhattan distance between any two points is maximized.

Return the maximum possible minimum Manhattan distance between the selected k points.

The Manhattan Distance between two cells (xi, yi) and (xj, yj) is |xi - xj| + |yi - yj|.

 

Example 1:

Input: side = 2, points = [[0,2],[2,0],[2,2],[0,0]], k = 4

Output: 2

Explanation:



Select all four points.

Example 2:

Input: side = 2, points = [[0,0],[1,2],[2,0],[2,2],[2,1]], k = 4

Output: 1

Explanation:



Select the points (0, 0), (2, 0), (2, 2), and (2, 1).

Example 3:

Input: side = 2, points = [[0,0],[0,1],[0,2],[1,2],[2,0],[2,2],[2,1]], k = 5

Output: 1

Explanation:



Select the points (0, 0), (0, 1), (0, 2), (1, 2), and (2, 2).

 

Constraints:

1 <= side <= 109
4 <= points.length <= min(4 * side, 15 * 103)
points[i] == [xi, yi]
The input is generated such that:
points[i] lies on the boundary of the square.
All points[i] are unique.
4 <= k <= min(25, points.length)


Solution 1:

class Solution:
    def maxDistance(self, side: int, points: List[List[int]], k: int) -> int:
        def get1D(side, x, y):
            """
            Map a boundary point (x, y) to its 1D position along the square perimeter.
            Imagine "unfolding" the square's perimeter into a straight line of length 4 * side.
            """
            # Bottom edge: distance is just the x-coordinate
            if y == 0:
                return x
            # Right edge: distance is the full bottom edge (side) + the y-coordinate
            if x == side:
                return side + y
            # Top edge: distance is bottom + right (3 * side) - distance from left to right
            if y == side:
                return 3 * side - x
            # Left edge: distance is bottom + right + top (4 * side) - distance from bottom to top
            return 4 * side - y

        def check(doubled, n, k, side, mid):
            """
            Check if we can pick 'k' points such that the minimum distance 
            between any two adjacent points (and the last to first) is at least 'mid'.
            """
            perimeter = 4 * side

            # Try starting our sequence of k points from every possible point i
            for i in range(n):  
                count = 1
                idx = i
                lastPos = doubled[idx]

                # Greedily try to find the remaining k - 1 points
                while count < k:
                    # The next point must be at least 'mid' distance away from the last chosen point
                    target = lastPos + mid
                    
                    # Use binary search to quickly find the first point >= target
                    # We search in the range [idx + 1, i + n] to avoid wrapping past our starting point
                    it = bisect.bisect_left(doubled, target, idx + 1, i + n)
                    
                    # If we couldn't find a valid next point before completing the circle, this starting point fails
                    if it == i + n:
                        break
                        
                    # Update our indices and positions for the next iteration
                    idx = it
                    lastPos = doubled[idx]
                    count += 1

                # If we successfully found k points, we need one final check:
                # Because it's a circle, the distance between the LAST point we picked 
                # and the FIRST point we picked must ALSO be at least 'mid'.
                # (doubled[i] + perimeter) represents the first point wrapped into the next cycle.
                if count == k and (doubled[i] + perimeter - lastPos >= mid):
                    return True

            # If no starting point 'i' allowed us to place 'k' points validly, return False
            return False

            
        perimeter = 4 * side
        n = len(points)

        # Step 1: Convert all 2D coordinates to 1D distances along the perimeter
        positions = [get1D(side, x, y) for x, y in points]
        
        # Sort them so we can process them sequentially and use binary search
        positions.sort()

        # Step 2: Double the array to easily handle the circular nature of the square.
        # Instead of using complex modulo arithmetic, we just append a second copy 
        # shifted by the total perimeter length.
        doubled = positions + [p + perimeter for p in positions]

        # Step 3: Binary search for the maximum feasible distance
        # The minimum possible distance is 0, maximum is half the perimeter (2 * side)
        l, r = 0, 2 * side
        result = 0

        while l <= r:
            mid = (l + r) // 2
            
            # If we can successfully place k points with at least 'mid' distance...
            if check(doubled, n, k, side, mid):
                result = mid    # Record this as our best answer so far
                l = mid + 1     # Try to find a LARGER distance
            else:
                r = mid - 1     # 'mid' was too large, try a SMALLER distance

        return result

class Solution:
    def maxDistance(self, side: int, points: List[List[int]], k: int) -> int:
        n = len(points)
        perimeter = 4 * side
        
        # Step 1: Map 2D boundary points to 1D positions along the perimeter
        pos = [0] * n
        for i in range(n):
            x, y = points[i]
            if y == 0:
                pos[i] = x
            elif x == side:
                pos[i] = side + y
            elif y == side:
                pos[i] = 3 * side - x
            else:
                pos[i] = 4 * side - y
                
        pos.sort()
        
        # Double the array to handle circular wrap-around seamlessly
        doubled = pos + [p + perimeter for p in pos]

        def check(dist: int) -> bool:
            """
            Checks if we can pick k points with a perimeter distance >= dist.
            """
            next_idx = [0] * (2 * n)
            j = 0
            
            # Precompute the next valid point for EVERY index using Two Pointers.
            # Because 'doubled' is sorted, 'j' only ever moves forward. This is O(N).
            for i in range(2 * n):
                while j < 2 * n and doubled[j] - doubled[i] < dist:
                    j += 1
                next_idx[i] = j

            # Try starting our k-point selection from each point in the original array
            for i in range(n):
                curr = i
                
                # Make k - 1 jumps using our precomputed O(1) jump table
                for _ in range(k - 1):
                    curr = next_idx[curr]
                    # If we run out of array space, this starting point fails.
                    # Furthermore, any starting point AFTER 'i' will also fail, so we can exit early!
                    if curr == 2 * n:
                        return False
                
                # Final check: Does the distance from the last point back to the first point >= dist?
                if doubled[i] + perimeter - doubled[curr] >= dist:
                    return True
                    
            return False

        # Binary search for the optimal distance
        # Minimum possible answer is 1, maximum is exactly 'side'
        l, r = 1, side
        ans = 1
        
        while l <= r:
            mid = (l + r) // 2
            if check(mid):
                ans = mid        # It's possible, save it and try for a larger distance
                l = mid + 1
            else:
                r = mid - 1      # Too large, search lower
                
        return ans
