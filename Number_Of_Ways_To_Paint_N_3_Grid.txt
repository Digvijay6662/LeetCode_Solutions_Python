You have a grid of size n x 3 and you want to paint each cell of the grid with exactly one of the three colors: Red, Yellow, or Green while making sure that no two adjacent cells have the same color (i.e., no two cells that share vertical or horizontal sides have the same color).

Given n the number of rows of the grid, return the number of ways you can paint this grid. As the answer may grow large, the answer must be computed modulo 109 + 7.

 

Example 1:


Input: n = 1
Output: 12
Explanation: There are 12 possible way to paint the grid as shown.
Example 2:

Input: n = 5000
Output: 30228214
 

Constraints:

n == grid.length
1 <= n <= 5000


Solution 1:

class Solution:
    def numOfWays(self, n: int) -> int:
        MOD = 10**9 + 7
        
        # All possible row patterns (some are invalid, but kept here to match your original approach)
        row_patterns = [
            "RYG", "RGY", "RYR", "RGR", 
            "YGR", "YRG", "YRY", "YGY", 
            "GRY", "GYR", "GRG", "GYG"
        ]

        # dictionary: stores results for (row_index, previous_row_pattern_index)
        dp = {}

        def countWays(row_index, prev_pattern_index):
            # Base case: if we've painted all rows, return 1 valid way
            if row_index == n:
                return 1
            
            # If already computed, return cached result
            if (row_index, prev_pattern_index) in dp:
                return dp[(row_index, prev_pattern_index)]
            
            total = 0
            prev_pattern = row_patterns[prev_pattern_index]

            # Try all possible current row patterns
            for curr_pattern_index in range(12):
                if curr_pattern_index == prev_pattern_index:
                    # Skip identical pattern (not necessary but keeps symmetry with your code)
                    continue

                curr_pattern = row_patterns[curr_pattern_index]
                conflict = False

                # Check vertical conflict: same column color in consecutive rows
                for col in range(3):
                    if curr_pattern[col] == prev_pattern[col]:
                        conflict = True
                        break

                # If no vertical conflict, recurse to next row
                if not conflict:
                    total = (total + countWays(row_index + 1, curr_pattern_index)) % MOD

            # Store result in dp table
            dp[(row_index, prev_pattern_index)] = total
            return total

        # Try all possible first-row patterns
        result = 0
        for first_pattern_index in range(12):
            result = (result + countWays(1, first_pattern_index)) % MOD

        return result


Solution 2:

class Solution:
    def numOfWays(self, n: int) -> int:
        MOD = 10**9 + 7  

        # For n = 1:
        # aba = 2-color patterns (e.g., RYR). There are 6: RYR, RGR, YRY, YGY, GRG, GYG.
        # abc = 3-color patterns (e.g., RYG). There are 6: RYG, RGY, YRG, YGR, GRY, GYR.
        aba, abc = 6, 6  

        # Build the solution row by row from row 2 to n
        for _ in range(2, n + 1):
            # Transition rules:
            # A 2-color row (ABA) can be followed by:
            #   - 3 ABA patterns (e.g., if RYR: YRY, YGY, GRG)
            #   - 2 ABC patterns (e.g., if RYR: YGR, GRY)
            #   New ABA = (aba * 3 + abc * 2)
            #
            # A 3-color row (ABC) can be followed by:
            #   - 2 ABA patterns (e.g., if RYG: YRY, GYG)
            #   - 2 ABC patterns (e.g., if RYG: YGR, GYR)
            #   New ABC = (aba * 2 + abc * 2)
            
            next_aba = (aba * 3 + abc * 2) % MOD
            next_abc = (aba * 2 + abc * 2) % MOD
            
            aba, abc = next_aba, next_abc

        return (aba + abc) % MOD
