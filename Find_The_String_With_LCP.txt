We define the lcp matrix of any 0-indexed string word of n lowercase English letters as an n x n grid such that:

lcp[i][j] is equal to the length of the longest common prefix between the substrings word[i,n-1] and word[j,n-1].
Given an n x n matrix lcp, return the alphabetically smallest string word that corresponds to lcp. If there is no such string, return an empty string.

A string a is lexicographically smaller than a string b (of the same length) if in the first position where a and b differ, string a has a letter that appears earlier in the alphabet than the corresponding letter in b. For example, "aabd" is lexicographically smaller than "aaca" because the first position they differ is at the third letter, and 'b' comes before 'c'.

 

Example 1:

Input: lcp = [[4,0,2,0],[0,3,0,1],[2,0,2,0],[0,1,0,1]]
Output: "abab"
Explanation: lcp corresponds to any 4 letter string with two alternating letters. The lexicographically smallest of them is "abab".
Example 2:

Input: lcp = [[4,3,2,1],[3,3,2,1],[2,2,2,1],[1,1,1,1]]
Output: "aaaa"
Explanation: lcp corresponds to any 4 letter string with a single distinct letter. The lexicographically smallest of them is "aaaa". 
Example 3:

Input: lcp = [[4,3,2,1],[3,3,2,1],[2,2,2,1],[1,1,1,3]]
Output: ""
Explanation: lcp[3][3] cannot be equal to 3 since word[3,...,3] consists of only a single letter; Thus, no answer exists.
 

Constraints:

1 <= n == lcp.length == lcp[i].length <= 1000
0 <= lcp[i][j] <= n


Solution 1:

class Solution:
    def findTheString(self, lcp: List[List[int]]) -> str:

        # Helper function to compute the LCP matrix of a given string
        def build_lcp_matrix(s):
            n = len(s)
            result = [[0] * n for _ in range(n)]
            
            # Fill last row and last column
            for j in range(n):
                result[n-1][j] = result[j][n-1] = 1 if s[j] == s[n-1] else 0
            
            # Fill the rest of the matrix from bottom-right upwards
            for i in range(n-2, -1, -1):
                for j in range(n-2, -1, -1):
                    if s[i] == s[j]:
                        result[i][j] = 1 + result[i+1][j+1]
                    else:
                        result[i][j] = 0
            return result

        n = len(lcp)
        candidate_chars = ["a"] * n   # Start with all 'a's

        # Try to assign characters one by one
        for pos in range(1, n):       # pos = current index we are filling
            matched = False
            forbidden_letters = [False] * 26  # Track which letters cannot be used

            # Compare with all previous positions
            for prev in range(pos):
                if lcp[prev][pos] == 0:
                    # If LCP is 0, then characters must differ
                    forbidden_letters[ord(candidate_chars[prev]) - ord('a')] = True
                    continue
                # Otherwise, they must match
                matched = True
                candidate_chars[pos] = candidate_chars[prev]
                break

            if matched:
                continue

            # If no match was forced, pick the smallest available letter
            for letter_index in range(26):
                if forbidden_letters[letter_index]:
                    continue
                candidate_chars[pos] = chr(ord('a') + letter_index)
                break

        # Build the candidate string
        candidate = "".join(candidate_chars)

        # Verify the candidate string actually matches the given LCP matrix
        if build_lcp_matrix(candidate) == lcp:
            return candidate
        return ""


Solution 2:

class Solution:
    def findTheString(self, lcp: List[List[int]]) -> str:
        n = len(lcp)
        # Initialize word with placeholders
        word = ["$"] * n   

        # Step 1: Construct the word character by character
        for i in range(n):
            # Try to match with a previous position if LCP > 0
            for j in range(i):
                if lcp[j][i] != 0:
                    word[i] = word[j]   # must be same character
                    break

            # If still unassigned, pick the smallest valid letter
            if word[i] == "$":
                forbidden = [False] * 26  # track letters that cannot be used
                for j in range(i):
                    if lcp[j][i] == 0:    # must differ from word[j]
                        forbidden[ord(word[j]) - ord('a')] = True

                # Assign the lexicographically smallest allowed letter
                for idx in range(26):
                    if not forbidden[idx]:
                        word[i] = chr(ord('a') + idx)
                        break

                # If no letter could be assigned, return failure
                if word[i] == "$":
                    return ""  

        # Step 2: Validate the constructed word against the given LCP matrix
        def checkLCP(candidate):
            n = len(candidate)

            # Check last row/column consistency
            for i in range(n):
                if candidate[i] != candidate[n-1]:
                    if lcp[i][n-1] != 0:
                        return False
                else:
                    if lcp[i][n-1] != 1:
                        return False

            for j in range(n):
                if candidate[n-1] != candidate[j]:
                    if lcp[n-1][j] != 0:
                        return False
                else:
                    if lcp[n-1][j] != 1:
                        return False

            # Check rest of the matrix
            for i in range(n-2, -1, -1):
                for j in range(n-2, -1, -1):
                    if candidate[i] == candidate[j]:
                        if lcp[i][j] != 1 + lcp[i+1][j+1]:
                            return False
                    else:
                        if lcp[i][j] != 0:
                            return False

            return True
        
        candidate = "".join(word)
        return candidate if checkLCP(candidate) else ""
