You are given an integer c representing c power stations, each with a unique identifier id from 1 to c (1‑based indexing).

These stations are interconnected via n bidirectional cables, represented by a 2D array connections, where each element connections[i] = [ui, vi] indicates a connection between station ui and station vi. Stations that are directly or indirectly connected form a power grid.

Initially, all stations are online (operational).

You are also given a 2D array queries, where each query is one of the following two types:

[1, x]: A maintenance check is requested for station x. If station x is online, it resolves the check by itself. If station x is offline, the check is resolved by the operational station with the smallest id in the same power grid as x. If no operational station exists in that grid, return -1.

[2, x]: Station x goes offline (i.e., it becomes non-operational).

Return an array of integers representing the results of each query of type [1, x] in the order they appear.

Note: The power grid preserves its structure; an offline (non‑operational) node remains part of its grid and taking it offline does not alter connectivity.

 

Example 1:

Input: c = 5, connections = [[1,2],[2,3],[3,4],[4,5]], queries = [[1,3],[2,1],[1,1],[2,2],[1,2]]

Output: [3,2,3]

Explanation:



Initially, all stations {1, 2, 3, 4, 5} are online and form a single power grid.
Query [1,3]: Station 3 is online, so the maintenance check is resolved by station 3.
Query [2,1]: Station 1 goes offline. The remaining online stations are {2, 3, 4, 5}.
Query [1,1]: Station 1 is offline, so the check is resolved by the operational station with the smallest id among {2, 3, 4, 5}, which is station 2.
Query [2,2]: Station 2 goes offline. The remaining online stations are {3, 4, 5}.
Query [1,2]: Station 2 is offline, so the check is resolved by the operational station with the smallest id among {3, 4, 5}, which is station 3.
Example 2:

Input: c = 3, connections = [], queries = [[1,1],[2,1],[1,1]]

Output: [1,-1]

Explanation:

There are no connections, so each station is its own isolated grid.
Query [1,1]: Station 1 is online in its isolated grid, so the maintenance check is resolved by station 1.
Query [2,1]: Station 1 goes offline.
Query [1,1]: Station 1 is offline and there are no other stations in its grid, so the result is -1.
 

Constraints:

1 <= c <= 105
0 <= n == connections.length <= min(105, c * (c - 1) / 2)
connections[i].length == 2
1 <= ui, vi <= c
ui != vi
1 <= queries.length <= 2 * 105
queries[i].length == 2
queries[i][0] is either 1 or 2.
1 <= queries[i][1] <= c


Solution 1:

class Solution:
    def processQueries(self, c: int, connections: List[List[int]], queries: List[List[int]]) -> List[int]:
        # Build adjacency list for the power grid connections
        graph = defaultdict(list)
        for u, v in connections:
            graph[u].append(v)
            graph[v].append(u)

        # Set to track which stations are currently online
        online_stations = set()

        # For each connected component (grid), maintain a min-heap of station IDs
        # This allows quick access to the smallest operational station ID
        grid_min_heap = defaultdict(list)

        # Map each station to its grid identifier
        station_to_grid = defaultdict(int)

        # DFS to assign stations to their grid and initialize heaps
        def dfs(station, grid_id):
            online_stations.add(station)                  # Mark station as online initially
            station_to_grid[station] = grid_id            # Assign station to its grid
            heapq.heappush(grid_min_heap[grid_id], station)  # Add station to grid's min-heap

            # Visit all neighbors in the same grid
            for neighbor in graph[station]:
                if neighbor not in online_stations:
                    dfs(neighbor, grid_id)

        # Identify all grids (connected components)
        for station in range(1, c + 1):
            if station not in online_stations:
                dfs(station, station)  # Use station ID itself as grid identifier

        results = []
        # Process each query
        for query_type, station in queries:
            if query_type == 1:  # Maintenance check
                if station in online_stations:
                    # If station is online, it resolves the check itself
                    results.append(station)
                else:
                    # If station is offline, find smallest online station in its grid
                    grid_id = station_to_grid[station]
                    min_heap = grid_min_heap[grid_id]

                    # Remove offline stations from heap top until valid
                    while min_heap and min_heap[0] not in online_stations:
                        heapq.heappop(min_heap)

                    if min_heap:
                        results.append(min_heap[0])  # Smallest operational station
                    else:
                        results.append(-1)           # No operational station in grid
            else:  # query_type == 2 → Station goes offline
                online_stations.discard(station)

        return results

class UnionFind:
    def __init__(self, n: int):
        # parent[i] = representative (root) of the set containing i
        self.parent = [i for i in range(n)]
        # size[i] = size of the set rooted at i (used for union by size)
        self.size = [1] * n

    def find(self, node: int) -> int:
        # Path compression: flatten the tree for efficiency
        while node != self.parent[node]:
            self.parent[node] = self.parent[self.parent[node]]
            node = self.parent[node]
        return node

    def union(self, node1: int, node2: int) -> bool:
        # Find roots of both nodes
        root1, root2 = self.find(node1), self.find(node2)
        if root1 == root2:
            return False  # already in the same set

        # Union by size: attach smaller tree under larger one
        if self.size[root1] > self.size[root2]:
            self.parent[root2] = root1
            self.size[root1] += self.size[root2]
        else:
            self.parent[root1] = root2
            self.size[root2] += self.size[root1]

        return True

class Solution:
    def processQueries(self, c: int, connections: List[List[int]], queries: List[List[int]]) -> List[int]:
        uf = UnionFind(c + 1)  # +1 because stations are 1-indexed

        # Step 1: Build connected components using Union-Find
        for u, v in connections:
            uf.union(u, v)

        # Step 2: Initialize heaps for each component
        # Each component root has a min-heap of station IDs
        component_to_heap = {}
        online_stations = set(range(1, c + 1))  # initially all stations are online

        for station in range(1, c + 1):
            root = uf.find(station)
            if root not in component_to_heap:
                component_to_heap[root] = []
            heapq.heappush(component_to_heap[root], station)

        # Step 3: Process queries
        results = []
        for query_type, station in queries:
            if query_type == 1:  # Maintenance check
                if station in online_stations:
                    # If station is online, it resolves the check itself
                    results.append(station)
                else:
                    # If station is offline, find smallest online station in its component
                    root = uf.find(station)
                    min_heap = component_to_heap[root]

                    # Lazy deletion: remove offline stations from heap top
                    while min_heap and min_heap[0] not in online_stations:
                        heapq.heappop(min_heap)

                    if min_heap:
                        results.append(min_heap[0])  # smallest online station
                    else:
                        results.append(-1)  # no online station in this component
            else:  # query_type == 2 → Station goes offline
                online_stations.discard(station)

        return results
