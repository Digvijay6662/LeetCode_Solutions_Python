On day 1, one person discovers a secret.

You are given an integer delay, which means that each person will share the secret with a new person every day, starting from delay days after discovering the secret. You are also given an integer forget, which means that each person will forget the secret forget days after discovering it. A person cannot share the secret on the same day they forgot it, or on any day afterwards.

Given an integer n, return the number of people who know the secret at the end of day n. Since the answer may be very large, return it modulo 109 + 7.

 

Example 1:

Input: n = 6, delay = 2, forget = 4
Output: 5
Explanation:
Day 1: Suppose the first person is named A. (1 person)
Day 2: A is the only person who knows the secret. (1 person)
Day 3: A shares the secret with a new person, B. (2 people)
Day 4: A shares the secret with a new person, C. (3 people)
Day 5: A forgets the secret, and B shares the secret with a new person, D. (3 people)
Day 6: B shares the secret with E, and C shares the secret with F. (5 people)
Example 2:

Input: n = 4, delay = 1, forget = 3
Output: 6
Explanation:
Day 1: The first person is named A. (1 person)
Day 2: A shares the secret with B. (2 people)
Day 3: A and B share the secret with 2 new people, C and D. (4 people)
Day 4: A forgets the secret. B, C, and D share the secret with 3 new people. (6 people)
 

Constraints:

2 <= n <= 1000
1 <= delay < forget <= n


Solution 1:

class Solution:
    def peopleAwareOfSecret(self, n: int, delay: int, forget: int) -> int:
        MOD = 10**9 + 7  # Large number modulus to prevent overflow
        
        # dp[i] = number of people who learn the secret on day i
        dp = [0] * (n + 1)
        dp[1] = 1  # On day 1, exactly 1 person knows the secret
        
        # Loop through each day starting from day 2
        for i in range(2, n + 1):
            # People who can share the secret today are those who learned it
            # between (i - forget + 1) and (i - delay) days ago
            #   - They haven't forgotten yet (forget days not passed)
            #   - They have waited at least 'delay' days before sharing
            for j in range(i - forget + 1, i - delay + 1):
                if j > 0:  # Ignore invalid days (before day 1)
                    dp[i] = (dp[i] + dp[j]) % MOD  # Add sharers from day j
        
        # Count how many people still remember the secret on day n
        total = 0
        # People who learned it in the last 'forget - 1' days still remember
        for i in range(n - forget + 1, n + 1):
            if i > 0:  # Ignore invalid days
                total = (total + dp[i]) % MOD
        
        return total


Solution 2:

class Solution:
    def peopleAwareOfSecret(self, n: int, delay: int, forget: int) -> int:
        MOD = 10**9 + 7
        dp = [0] * (n + 1)
        dp[1] = 1  # Day 1: one person learns the secret
        
        sharers = 0  # Number of people who can share today
        
        for day in range(2, n + 1):
            # People who start sharing today
            if day - delay > 0:
                sharers = (sharers + dp[day - delay]) % MOD
            
            # People who forget today (stop sharing)
            if day - forget > 0:
                sharers = (sharers - dp[day - forget]) % MOD
            
            # New people who learn the secret today
            dp[day] = sharers % MOD
        
        # Count people who still remember at the end
        return sum(dp[max(1, n - forget + 1): n + 1]) % MOD


Solution 3:

class Solution:
    def peopleAwareOfSecret(self, n: int, delay: int, forget: int) -> int:
        MOD = 10**9 + 7
        
        # Queue will store: number of people who learned the secret on each day
        # Index 0 = people who learned today, index 1 = yesterday, etc.
        q = deque([1])  # Day 1: 1 person knows
        total_people = 1  # Total people who currently remember the secret
        sharers = 0       # People who can share today
        
        for day in range(2, n + 1):
            # People who start sharing today
            if day - delay > 0:
                sharers = (sharers + q[delay - 1]) % MOD
            
            # People who forget today
            if day - forget > 0:
                sharers = (sharers - q[forget - 1]) % MOD
                total_people = (total_people - q[forget - 1]) % MOD
            
            # New people who learn today = current sharers
            new_people = sharers % MOD
            q.appendleft(new_people)  # Add today's learners to the front
            total_people = (total_people + new_people) % MOD
            
            # Keep queue size = forget days (older entries are forgotten)
            if len(q) > forget:
                q.pop()
        
        return total_people % MOD
